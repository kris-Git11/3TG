using namespace std;


struct drzave {
	string imeD[5], imeV[5];
	int pov[5], brojS[5], visinaV[5];
	
};
void sortiranje (int A[], int N) {
	int i, j, min, zamjeni;
	for (i = 0; i < N; i++) {
		min = i;
		for (j = i + 1; j < N; j++) {
			if (A[j] < A[min]) min = j;
		}
		zamjeni = A[i];
		A[i] = A[min];
		A[min] = zamjeni;
	}
}
