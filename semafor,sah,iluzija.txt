private void Form1_Paint(object sender, PaintEventArgs e)
{
Graphics g = e.Graphics;
            Pen zelenaOlovka = new Pen(Color.Green);
            Pen crvenaOlovka = new Pen(Color.PaleVioletRed);
            Pen plavaOlovka = new Pen(Color.DarkBlue);
            Pen crnaOlovka = new Pen(Color.Black);

            SolidBrush zeleniKist = new SolidBrush(Color.Green);
            SolidBrush zutiKist = new SolidBrush(Color.Yellow);
            SolidBrush crvenaKist = new SolidBrush(Color.Red);
            SolidBrush crniKist = new SolidBrush(Color.Black);

            Rectangle r1 = new Rectangle(50, 50, 50, 50);
            Rectangle r2 = new Rectangle(50, 100, 50, 50);
            Rectangle r3 = new Rectangle(50, 150, 50, 50);

            g.DrawRectangle(plavaOlovka, r1);
            g.DrawRectangle(crnaOlovka, r2);
            g.DrawRectangle(crnaOlovka, r3);

            g.FillRectangle(crniKist, r1);
            g.FillRectangle(crniKist, r2);
            g.FillRectangle(crniKist, r3);

            g.FillEllipse(crvenaKist, r1);
            g.FillEllipse(zutiKist, r2);
            g.FillEllipse(zeleniKist, r3);

            zelenaOlovka.Dispose();
            crvenaOlovka.Dispose();
            plavaOlovka.Dispose();
            g.Dispose();
}

private void Form1_Paint(object sender, PaintEventArgs e)
        {
            Graphics g = e.Graphics;

            for (int i=1;i<9;i++)
            {
                for (int j=1;j<9;j++)
                {
                    Pen blackPen = new Pen(Color.Black, 1);
                    Rectangle r1 = new Rectangle(50 * i, 50 * j, 50, 50);
                    g.DrawRectangle(blackPen, r1);
                    if ((i + j) % 2 == 0)
                    {
                        SolidBrush smediKist = new SolidBrush(Color.Brown);
                        g.FillRectangle(smediKist, r1);
                    }
                    if (j == 2 ) 
                    {
                        SolidBrush crniKist = new SolidBrush(Color.Black);
                        g.FillEllipse(crniKist, r1);
                    }
                    if (j == 7)
                    {
                        Pen crnaOlovka = new Pen(Color.Black);
                        SolidBrush bijeliKist = new SolidBrush(Color.White);
                        g.FillEllipse(bijeliKist, r1);
                        g.DrawEllipse(crnaOlovka, r1);
                    }
                    
                }
            }   
        }
